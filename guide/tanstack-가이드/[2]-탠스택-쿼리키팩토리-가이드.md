# 1. QueryKey 상수로 생성
```
export const QUERY_KEYS = {
  todo: {
    all: ['todo'],
    list: ['todo', 'list'],
    detail: (id: string) => ["todo", "detail", id]
  }
}
```

# 2. 조회시 사용하는 쿼리키 예시
```
  const { data, isError, isLoading } = useQuery({
    queryFn: fetchTodos,
    queryKey: QUERY_KEYS.todo.list //✅ 여기 이렇게 호출
  })
```

# 3. insert시 무효화 쿼리키 예시
```
  const { mutate, isPending } = useMutation({
    mutationFn: (content: string) => insertTodo(content),
    onSuccess: () => {
      queryClient.invalidateQueries({
        queryKey: QUERY_KEYS.todo.list // ✅ 여기
      })
      toast('추가 완료');
    },
    onError: (err) => {
      toast(err.message);
    }
  })
```
