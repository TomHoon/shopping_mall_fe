# update 통신 완료 전에 통신이 성공했다고 가정하고 TanStack 캐시 데이터를 먼저 업데이트 시킵니다.

```

  const {
    mutate: updateMutate
  } = useMutation({

    mutationFn: ({id, isDone}: {id: string, isDone: boolean}) => updateTodo(id, isDone),

    mutationKey: QUERY_KEYS.todo.list,

    onMutate: async (updatedTodo) => {

      // ✅ 1. 조회가 다 되기 전에 업데이트를 막기 위한 함수
      // 조회 중인데 Update를 날린다 -> 조회가 완료되어 데이터가 셋팅된다 -> 업데이트 된 녀석과 맞지않는다.
      await queryClient.cancelQueries({queryKey: QUERY_KEYS.todo.list});

      // ✅ 2. 에러날 경우를 대비해 스냅샷을 찍어 데이터를 원복하기 위함
      const previous = queryClient.getQueryData(QUERY_KEYS.todo.list);

      // ✅ 3. 낙관적 업데이트 시작
      queryClient.setQueryData<Todo[]>(
        QUERY_KEYS.todo.list,

        (prevData) => {
          if (!prevData) return [];

          return prevData.map(item => ({
            ...item,
            isDone: item.id === id ? !item.isDone : item.isDone
          }))
        }

      )

      // ✅ 스냅샷 리턴(5번에서 사용할 녀석)
      return {previous};
    },
    // ✅ 4. 에러가 날 경우 2번에서 가져온 스냅샷을 사용함(context에 들어온다. )
    onError: (error, variable, context) => {
      queryClient.setQueryData(QUERY_KEYS.todo.list,
        context?.previous
      );
    },
    // ✅ 5. 업데이트 다 된 후 캐시 초기화를 통해 실제 데이터를 다시 셋팅함
    // 즉각적 UX를 먼저 제공하고 실제 데이터는 나중에 조회하는 방식임
    onSettled: () => {
      queryClient.invalidateQueries({queryKey: QUERY_KEYS.todo.list})
    }

  })

```
