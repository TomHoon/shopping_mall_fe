# 리스트 내에는 상세 데이터들이 포함되어 있다. 리스트를 캐시하지 않고 각각 디테일 데이터를 캐시하는 편이 더 유리하다.

# 1. 추가
> 디테일 캐시에 데이터 추가하기
> 리스트 캐시에 id 값 추가하기
```
  const { mutate, isPending } = useMutation({
    mutationFn: (content: string) => insertTodo(content),
    onSuccess: (newData) => {
      // ✅ 캐시 리스트에 추가하는 버전
      queryClient.setQueryData<Todo[]>(
        QUERY_KEYS.todo.list,
        (ids) => {
          if (!ids) return [newData.id];
          return [...ids, newData.id]
        }
      );

      queryClient.setQueryData<Todo>(
        QUERY_KEYS.todo.detail(newData.id),
        (prev) => {
          return {
            ...newData
          }
        }
      )

      toast('추가 완료');
    },
    onError: (err) => {
      toast(err.message);
    }
  })
```

# 2. 수정
> 기존 디테일 캐시에 값 수정하기
> 리스트 변화 없음
```
  // update...
  const {
    mutate: updateMutate
  } = useMutation({
    mutationFn: ({id, isDone}: {id: string, isDone: boolean}) => updateTodo(id, isDone),
    mutationKey: QUERY_KEYS.todo.list,
    onMutate: async (updatedTodo) => {
      // 바뀌고 나서 데이터임

      // ✅ 1. 수정하기 이전에 조회 pending 있으면 제거해줌
      await queryClient.cancelQueries({queryKey: QUERY_KEYS.todo.list});
      // ✅ 2. 실패시 원복을 위해 스냅샷 저장
      const previous = queryClient.getQueryData(QUERY_KEYS.todo.list);


      // ✅ 3. 해당하는 디테일 캐시값만 수정한다. (낙관적 업데이트)
      queryClient.setQueryData<Todo>(
        QUERY_KEYS.todo.detail(updatedTodo.id),
        (prev) => {
          if (!prev) return;
          
          return {
            ...prev,
            isDone: updatedTodo.isDone
          }
        }
      )

      return {previous};
    },
    onError: (error, variable, context) => {

      // ✅ 4. 실패시 원복함
      queryClient.setQueryData(
        QUERY_KEYS.todo.list,
        context?.previous
      );
    },
    onSettled: () => {
      // ✅ 5. 성공시 데이터 저장함
      queryClient.invalidateQueries({queryKey: QUERY_KEYS.todo.list})
    }

  })
```

    const data = await res.json();
    return data;
